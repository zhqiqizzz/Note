```bash
# 查看服务状态
sudo systemctl status easytier@default
sudo systemctl status easytier-route.service

# 查看节点连接
sudo easytier-cli peer

# 查看路由和策略
ip route show
ip rule show

# 查看日志
sudo journalctl -u easytier@default -f

# 重启服务
sudo systemctl restart easytier@default
sudo systemctl restart easytier-route.service
```
