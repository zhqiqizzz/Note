### 树叶刮风影响树叶掉落，最后掉完叶子要多少个月。

每个月按比例掉，风会加快掉落，比如：

- 初始叶子数 `1000`
- 基础掉落率 `10%`
- 风力系数 `1.5`
- 实际每月掉落率 = `10% × 1.5 = 15%`

因为按比例掉时，理论上很难“刚好等于 0”，所以通常设一个阈值，比如剩下不到 `1` 片就算掉完。

```js
function monthsToFallAll_percent(totalLeaves, baseRate, windFactor, threshold = 1) {  
  if (totalLeaves <= threshold) return 0;  
  
  let rate = baseRate * windFactor;  
  if (rate <= 0) return Infinity;  
  if (rate >= 1) return 1; // 一个月内全掉完  
  
  let months = 0;  
  let leaves = totalLeaves;  
  
  while (leaves > threshold) {  
    leaves *= (1 - rate);  
    months++;  
  }  
  
  return months;  
} 
  
console.log(monthsToFallAll_percent(1000, 0.1, 1.5)); // 44
```

### 数组元素和，染红元素和，剩下元素和是多少。

可以直接这样算：

- **数组总和**
- **染红元素的和**
- **剩下元素的和 = 总和 - 染红元素的和**

假设：

```js
const arr = [3, 5, 2, 8, 6];  
const redIndex = [1, 3]; // 染红的是下标 1 和 3，对应 5 和 8
```

#### 写法 1：先算总和，再算染红和

```js
const arr = [3, 5, 2, 8, 6];  
const redIndex = [1, 3];  
  
const totalSum = arr.reduce((sum, num) => sum + num, 0);  
  
const redSum = redIndex.reduce((sum, i) => sum + arr[i], 0);  
  
const restSum = totalSum - redSum;  
  
console.log("数组总和:", totalSum);   // 24  
console.log("染红元素和:", redSum);   // 13  
console.log("剩下元素和:", restSum);  // 11
```

#### 写法 2：一次遍历
```js
const arr = [3, 5, 2, 8, 6];  
const redIndex = new Set([1, 3]);  
  
let redSum = 0;  
let restSum = 0;  
  
for (let i = 0; i < arr.length; i++) {  
  if (redIndex.has(i)) {  
    redSum += arr[i];  
  } else {  
    restSum += arr[i];  
  }  
}  
  
const totalSum = redSum + restSum;  
  
console.log("数组总和:", totalSum);   // 24  
console.log("染红元素和:", redSum);   // 13  
console.log("剩下元素和:", restSum);  // 11
```

### 文法分析，输出式子中合法格式式子的字符个数。

```js
function countValidExprChars(s) {
  let maxLen = 0;
  let curLen = 0;
  let validPrefix = false; // 当前是否处于一个“合法前缀”里
  let expectDigit = true;  // 当前是否必须读到数字

  for (const ch of s) {
    const isDigit = ch >= '0' && ch <= '9';
    const isOp = ch === '+' || ch === '-';

    if (!isDigit && !isOp) {
      // 遇到非法字符，直接断开
      curLen = 0;
      validPrefix = false;
      expectDigit = true;
      continue;
    }

    if (isDigit) {
      if (validPrefix) {
        curLen++;
      } else {
        // 重新开始一个新式子
        curLen = 1;
        validPrefix = true;
      }
      expectDigit = false;
      maxLen = Math.max(maxLen, curLen); // 只有以数字结尾才算完整合法式子
    } else {
      // 运算符
      if (validPrefix && !expectDigit) {
        curLen++;
        expectDigit = true;
      } else {
        // 运算符位置不合法，断开
        curLen = 0;
        validPrefix = false;
        expectDigit = true;
      }
    }
  }

  return maxLen;
}

// 测试
console.log(countValidExprChars("12+34-5"));   // 7
console.log(countValidExprChars("12++34"));    // 2
console.log(countValidExprChars("-12+3"));     // 4
console.log(countValidExprChars("1-2-"));      // 3
console.log(countValidExprChars("123"));       // 3
```
